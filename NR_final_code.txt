#include <stdio.h>
#include <math.h>

double f(double x)
 {
    return log(x) - x + 3;
}

double df(double x) 
{
    return 1 / x - 1;
}

int main() {
    double x, next, tol, error;
    int max_iter, i;

    printf("Enter initial guess (x0): ");
    scanf("%lf", &x);
    printf("Enter tolerance: ");
    scanf("%lf", &tol);
    printf("Enter maximum iterations: ");
    scanf("%d", &max_iter);

    for (i = 1; i <= max_iter; i++) 
    {
        next = x - f(x) / df(x);
        error = fabs(next - x);

        printf("Iteration %d: Approx root = %.6lf\n", i, next);

        if (error < tol) 
{
            printf("\nFinal root â‰ˆ %.6lf found in %d iterations.\n", next, i);
            return 0;
        }

        x = next;
    }

    printf("\nDid not converge within %d iterations.\n", max_iter);
    return 0;
}
